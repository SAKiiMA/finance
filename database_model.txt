CREATE TABLE IF NOT EXISTS 'users' ('id' INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, 'username' TEXT NOT NULL, 'hash' TEXT NOT NULL, 'cash' NUMERIC NOT NULL DEFAULT 10000.00 );

CREATE TABLE sqlite_sequence(name,seq);

CREATE UNIQUE INDEX 'username' ON "users" ("username");

CREATE TABLE IF NOT EXISTS 'transactions'(
'trans_id' INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
'user_id' NUMERIC NOT NULL,
'symbol' TEXT NOT NULL,
'company' TEXT NOT NULL,
'shares' NUMERIC NOT NULL,
'price' REAL NOT NULL,
'dated' DATETIME DEFAULT CURRENT_TIMESTAMP);

CREATE TABLE IF NOT EXISTS 'deposit' (
'topup_id' INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
'user_id' NUMERIC NOT NULL,
'amount' NUMERIC NOT NULL,
'dated' DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP);
